// phone case plus button event handler const casePlusBtn = document.getElementById("casePlus"); casePlusBtn.addEventListener("click", function(){ const phoneCase = document.getElementById("phoneCase").value; const itemNumber = parseFloat(phoneCase); const totalItem = itemNumber + 1; document.getElementById("phoneCase").value = totalItem; const totalPrice = totalItem * 59; document.getElementById("phoneCasePrice").innerText = totalPrice }) //phone case minus button event handler const caseMinusBtn = document.getElementById("caseMinus"); caseMinusBtn.addEventListener("click", function(){ const phoneCase = document.getElementById("phoneCase").value; const itemNumber = parseFloat(phoneCase); const totalItem = itemNumber - 1; document.getElementById("phoneCase").value = totalItem; const totalPrice = totalItem * 59; document.getElementById("phoneCasePrice").innerText = totalPrice })
